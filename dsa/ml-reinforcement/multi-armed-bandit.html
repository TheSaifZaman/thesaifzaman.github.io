Multi-Armed Bandit

Easy Time: O(k · rounds) Space: O(k)

How It Works

The Multi-Armed Bandit problem is the simplest form of reinforcement learning. You face k slot machines (arms), each with an unknown reward distribution. At each round, you pull one arm and receive a reward. The goal is to maximize total reward over many rounds.

The core challenge is the exploration-exploitation dilemma: should you try new arms to discover their rewards (explore), or should you keep pulling the arm that has given the best rewards so far (exploit)?

Common strategies include ε-greedy (random exploration with probability ε), Upper Confidence Bound (UCB, which favors uncertain arms), and Thompson Sampling (Bayesian approach using posterior distributions).

Exploration

Try unknown arms to learn their reward distributions.

Exploitation

Pull the best-known arm to maximize immediate reward.

UCB Strategy

Adds confidence bonus favoring less-explored arms.

Applications

A/B testing, ad placement, recommendation systems.

function epsilonGreedyBandit(k, rounds, epsilon) {
    var Q = new Array(k).fill(0);     // estimated values
    var N = new Array(k).fill(0);     // pull counts
    var totalReward = 0;

    for (var t = 0; t < rounds; t++) {
        var arm;
        if (Math.random() < epsilon) {
            arm = Math.floor(Math.random() * k); // explore
        } else {
            arm = argMax(Q);                       // exploit
        }

        var reward = pullArm(arm); // sample from arm's distribution
        N[arm]++;
        // Incremental mean update
        Q[arm] += (reward - Q[arm]) / N[arm];
        totalReward += reward;
    }
    return { Q: Q, totalReward: totalReward };
}

function ucbBandit(k, rounds, c) {
    var Q = new Array(k).fill(0);
    var N = new Array(k).fill(0);

    for (var t = 0; t < rounds; t++) {
        var arm;
        // Pull each arm once first
        if (t < k) {
            arm = t;
        } else {
            // UCB selection
            var best = -Infinity;
            for (var i = 0; i < k; i++) {
                var ucb = Q[i] + c * Math.sqrt(Math.log(t) / N[i]);
                if (ucb > best) { best = ucb; arm = i; }
            }
        }
        var reward = pullArm(arm);
        N[arm]++;
        Q[arm] += (reward - Q[arm]) / N[arm];
    }
    return Q;
}
