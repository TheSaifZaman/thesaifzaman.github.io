Suffix Array

Hard Build: O(n log n) Search: O(m log n) Space: O(n)

How It Works

A suffix array is a sorted array of all suffixes of a string. For a string of length n, the suffix starting at position i is S[i..n-1]. The suffix array stores the starting indices of these suffixes in lexicographic order. It provides a space-efficient alternative to suffix trees for many string processing tasks.

The naive construction sorts all n suffixes, taking O(n² log n) due to O(n) comparison cost. Efficient algorithms (like the prefix doubling approach by Karp-Miller-Rosenberg) achieve O(n log n) by iteratively sorting suffixes based on their first 1, 2, 4, 8, ... characters, using the previous round's ranks to compare in O(1).

Combined with the LCP (Longest Common Prefix) array, suffix arrays enable efficient substring search in O(m log n), finding longest repeated substrings, counting distinct substrings, and many other string operations. They use less memory than suffix trees while supporting similar queries.

Sorted Suffixes

All suffixes sorted lexicographically, stored as starting indices.

Space Efficient

function buildSuffixArray(s) {
    var n = s.length;
    var suffixes = [];
    for (var i = 0; i < n; i++) {
        suffixes.push({ index: i, suffix: s.substring(i) });
    }
    suffixes.sort(function (a, b) {
        return a.suffix < b.suffix ? -1 : a.suffix > b.suffix ? 1 : 0;
    });
    return suffixes.map(function (s) { return s.index; });
}

// O(n log n) prefix doubling approach
function buildSuffixArrayFast(s) {
    var n = s.length;
    var sa = [], rank = [], tmp = [];
    for (var i = 0; i < n; i++) {
        sa[i] = i;
        rank[i] = s.charCodeAt(i);
    }
    for (var k = 1; k < n; k *= 2) {
        var r = rank.slice();
        sa.sort(function (a, b) {
            if (r[a] !== r[b]) return r[a] - r[b];
            var ra = a + k < n ? r[a + k] : -1;
            var rb = b + k < n ? r[b + k] : -1;
            return ra - rb;
        });
        rank[sa[0]] = 0;
        for (var i = 1; i < n; i++) {
            rank[sa[i]] = rank[sa[i - 1]];
            if (r[sa[i]] !== r[sa[i - 1]] ||
                (sa[i] + k < n ? r[sa[i] + k] : -1) !==
                (sa[i - 1] + k < n ? r[sa[i - 1] + k] : -1)) {
                rank[sa[i]]++;
            }
        }
    }
    return sa;
}

function searchPattern(text, sa, pattern) {
    var lo = 0, hi = sa.length - 1;
    while (lo <= hi) {
        var mid = (lo + hi) >> 1;
        var suffix = text.substring(sa[mid], sa[mid] + pattern.length);
        if (suffix < pattern) lo = mid + 1;
        else if (suffix > pattern) hi = mid - 1;
        else return sa[mid]; // found
    }
    return -1;
}
