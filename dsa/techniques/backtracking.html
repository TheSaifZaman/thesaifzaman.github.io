Backtracking

Medium Time: O(n!) Space: O(n)

How It Works

Backtracking is a systematic method for exploring all possible configurations of a solution space. It builds candidates incrementally, abandoning a candidate (“backtracking”) as soon as it determines the candidate cannot lead to a valid solution.

The algorithm follows a depth-first search pattern through the decision tree. At each node, it makes a choice, recurses deeper, then undoes the choice before trying the next option. This “choose, explore, unchoose” pattern is the signature of backtracking.

Pruning is the key to making backtracking efficient. By detecting invalid partial solutions early, we avoid exploring large portions of the search tree. Without pruning, backtracking degenerates into brute-force enumeration.

Choose-Explore-Unchoose
Make a choice, recurse to explore, then undo the choice to try alternatives.
Pruning
Skip branches early when a partial solution cannot lead to a valid complete solution.
Decision Tree
The search space is a tree where each level represents a decision point.
Base Case
When the solution is complete (or invalid), stop recursing and record or reject.

// Permutations (LeetCode #46)
function permute(nums) {
    var result = [];
    function backtrack(path, used) {
        if (path.length === nums.length) {
            result.push(path.slice());
            return;
        }
        for (var i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            path.push(nums[i]);
            used[i] = true;
            backtrack(path, used);
            path.pop();       // unchoose
            used[i] = false;  // unchoose
        }
    }
    backtrack([], new Array(nums.length).fill(false));
    return result;
}

// Subsets (LeetCode #78)
function subsets(nums) {
    var result = [];
    function backtrack(start, path) {
        result.push(path.slice());
        for (var i = start; i < nums.length; i++) {
            path.push(nums[i]);
            backtrack(i + 1, path);
            path.pop();
        }
    }
    backtrack(0, []);
    return result;
}

// N-Queens (LeetCode #51)
function solveNQueens(n) {
    var result = [], board = [];
    for (var i = 0; i < n; i++) board.push(new Array(n).fill('.'));
    function isValid(row, col) {
        for (var i = 0; i < row; i++) {
            if (board[i][col] === 'Q') return false;
            if (col-(row-i) >= 0 && board[i][col-(row-i)] === 'Q') return false;
            if (col+(row-i) < n && board[i][col+(row-i)] === 'Q') return false;
        }
        return true;
    }
    function backtrack(row) {
        if (row === n) { result.push(board.map(function(r) { return r.join(''); })); return; }
        for (var col = 0; col < n; col++) {
            if (!isValid(row, col)) continue;
            board[row][col] = 'Q';
            backtrack(row + 1);
            board[row][col] = '.';
        }
    }
    backtrack(0);
    return result;
}
