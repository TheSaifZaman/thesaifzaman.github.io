Sliding Window

Medium Time: O(n) Space: O(k)

How It Works

The sliding window technique maintains a contiguous subarray or substring that expands and contracts as it slides through the data. It transforms problems that would require O(n²) nested loops into O(n) single-pass solutions by reusing computation from the previous window position.

There are two main variants: fixed-size windows (find the maximum sum of k consecutive elements) and variable-size windows (find the smallest window satisfying a condition). In the variable variant, expand the right boundary until the condition is met, then shrink from the left to find the optimal window.

The key invariant is that the window always represents a valid or candidate subarray. A hash map or counter often tracks the window contents, updated in O(1) as elements enter and leave the window.

Fixed Window
Window size k is constant. Slide by adding one element on the right and removing one on the left.
Variable Window
Expand right to satisfy a condition, then shrink left to optimize. Track the best result found.
Hash Map Tracking
Use a frequency map to track characters or values inside the window in O(1) per update.
Amortized O(n)
Each element enters the window once and leaves once, giving O(2n) = O(n) total operations.

// Maximum Sum of Subarray of Size K
function maxSubarraySum(arr, k) {
    var windowSum = 0, maxSum = 0;
    for (var i = 0; i < k; i++) windowSum += arr[i];
    maxSum = windowSum;
    for (var i = k; i < arr.length; i++) {
        windowSum += arr[i] - arr[i - k];
        maxSum = Math.max(maxSum, windowSum);
    }
    return maxSum;
}

// Longest Substring Without Repeating Characters (LeetCode #3)
function lengthOfLongestSubstring(s) {
    var map = {}, left = 0, best = 0;
    for (var right = 0; right < s.length; right++) {
        var ch = s[right];
        if (map[ch] !== undefined && map[ch] >= left) {
            left = map[ch] + 1;
        }
        map[ch] = right;
        best = Math.max(best, right - left + 1);
    }
    return best;
}

// Minimum Window Substring (LeetCode #76)
function minWindow(s, t) {
    var need = {}, have = {}, formed = 0, required = 0;
    for (var i = 0; i < t.length; i++) {
        need[t[i]] = (need[t[i]] || 0) + 1;
    }
    required = Object.keys(need).length;
    var left = 0, best = [Infinity, 0, 0];
    for (var right = 0; right < s.length; right++) {
        var ch = s[right];
        have[ch] = (have[ch] || 0) + 1;
        if (need[ch] && have[ch] === need[ch]) formed++;
        while (formed === required) {
            if (right - left + 1 < best[0]) best = [right - left + 1, left, right];
            var lch = s[left];
            have[lch]--;
            if (need[lch] && have[lch] < need[lch]) formed--;
            left++;
        }
    }
    return best[0] === Infinity ? "" : s.substring(best[1], best[2] + 1);
}
