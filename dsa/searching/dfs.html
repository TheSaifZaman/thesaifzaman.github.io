Depth-First Search

Explores as deep as possible before backtracking — O(V+E)

Concept

Depth-First Search (DFS) starts at a source node and explores each branch as far as possible before backtracking. It uses a stack (or recursion) to track which node to visit next.

  • Time Complexity: O(V + E)
  • Space Complexity: O(V) for the visited set and recursion stack
  • Uses: Cycle detection, topological sort, path finding, connected components
  • Traversal order: Goes deep into one branch before exploring siblings

Implementation

function dfs(graph, start) {
  const visited = new Set();
  const result = [];

  function traverse(node) {
    if (visited.has(node)) return;
    visited.add(node);
    result.push(node);

    for (const neighbor of graph[node]) {
      traverse(neighbor);
    }
  }

  traverse(start);
  return result;
}

// Iterative version using explicit stack
function dfsIterative(graph, start) {
  const visited = new Set();
  const stack = [start];
  const result = [];

  while (stack.length > 0) {
    const node = stack.pop();
    if (visited.has(node)) continue;

    visited.add(node);
    result.push(node);

    // Push neighbors in reverse for left-to-right order
    for (let i = graph[node].length - 1; i >= 0; i--) {
      stack.push(graph[node][i]);
    }
  }
  return result;
}
