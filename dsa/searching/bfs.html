Breadth-First Search

Explores all neighbors at current depth before going deeper — O(V+E)

Concept

Breadth-First Search (BFS) starts at a source node and explores all neighbors at the current depth level before moving to nodes at the next level. It uses a queue to maintain the frontier.

  • Time Complexity: O(V + E)
  • Space Complexity: O(V) for the queue and visited set
  • Uses: Shortest path (unweighted), level-order traversal, connected components
  • Key property: Visits nodes in order of their distance from the source

Implementation

function bfs(graph, start) {
  const visited = new Set([start]);
  const queue = [start];
  const result = [];

  while (queue.length > 0) {
    const node = queue.shift();
    result.push(node);

    for (const neighbor of graph[node]) {
      if (!visited.has(neighbor)) {
        visited.add(neighbor);
        queue.push(neighbor);
      }
    }
  }

  return result;
}

// BFS with level tracking
function bfsLevels(graph, start) {
  const visited = new Set([start]);
  let currentLevel = [start];
  const levels = [];

  while (currentLevel.length > 0) {
    levels.push([...currentLevel]);
    const nextLevel = [];

    for (const node of currentLevel) {
      for (const neighbor of graph[node]) {
        if (!visited.has(neighbor)) {
          visited.add(neighbor);
          nextLevel.push(neighbor);
        }
      }
    }
    currentLevel = nextLevel;
  }
  return levels;
}
