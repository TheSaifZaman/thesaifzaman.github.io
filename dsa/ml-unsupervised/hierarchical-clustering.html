Hierarchical Clustering

Medium Time: O(n³) Space: O(n²)

How It Works

Hierarchical Clustering builds a hierarchy of clusters using either a bottom-up (agglomerative) or top-down (divisive) approach. The agglomerative method starts with each point as its own cluster and repeatedly merges the two closest clusters until only one remains.

The distance between clusters can be measured using single linkage (minimum distance), complete linkage (maximum distance), or average linkage. The result is a dendrogram showing the merge history, which can be cut at any level to produce the desired number of clusters.

Key Properties

  • No k required upfront: Dendrogram can be cut at any level.
  • Deterministic: Same input always produces same dendrogram.
  • Linkage choice: Single, complete, average, or Ward's method.
  • Not scalable: O(n^3) time makes it slow for large datasets.

Implementation

function agglomerativeClustering(points) {
    // Start: each point is its own cluster
    var clusters = points.map(function(p, i) {
        return { id: i, points: [p] };
    });
    var history = [];

    while (clusters.length > 1) {
        var minDist = Infinity, mergeA = 0, mergeB = 1;

        // Find closest pair (single linkage)
        for (var i = 0; i < clusters.length; i++) {
            for (var j = i + 1; j < clusters.length; j++) {
                var d = clusterDist(clusters[i], clusters[j]);
                if (d < minDist) {
                    minDist = d; mergeA = i; mergeB = j;
                }
            }
        }

        // Merge
        var merged = {
            id: clusters[mergeA].id + '+' + clusters[mergeB].id,
            points: clusters[mergeA].points.concat(
                clusters[mergeB].points
            )
        };
        history.push({ a: mergeA, b: mergeB, dist: minDist });
        clusters.splice(mergeB, 1);
        clusters.splice(mergeA, 1, merged);
    }
    return history;
}

function clusterDist(a, b) {
    var min = Infinity;
    a.points.forEach(function(p1) {
        b.points.forEach(function(p2) {
            var d = Math.sqrt(
                Math.pow(p1.x-p2.x,2) + Math.pow(p1.y-p2.y,2)
            );
            if (d < min) min = d;
        });
    });
    return min;
}
