K-Medoids (PAM)

Medium Time: O(n²·k) Space: O(n·k)

How It Works

K-Medoids is similar to K-Means but uses actual data points as cluster centers (medoids) instead of computed means. The PAM (Partitioning Around Medoids) algorithm selects k data points as medoids, assigns each point to its nearest medoid, then tries swapping medoids with non-medoid points to minimize total dissimilarity.

Unlike K-Means, K-Medoids is more robust to outliers because medoids are actual data points. The swap step evaluates whether replacing a current medoid with another point reduces the total cost, making it more computationally expensive but producing more interpretable centers.

Key Properties

  • Robust: Less sensitive to outliers than K-Means.
  • Interpretable: Centers are actual data points.
  • Any metric: Works with any distance function, not just Euclidean.
  • Slower: O(n^2*k) vs O(n*k*i) for K-Means.

Implementation

function kMedoids(points, k, maxIter) {
    var n = points.length;
    // Initialize: pick k random points as medoids
    var medoids = [0, 5, 10]; // indices

    function totalCost(medoids) {
        var cost = 0;
        points.forEach(function(p, i) {
            var minD = Infinity;
            medoids.forEach(function(m) {
                var d = Math.sqrt(
                    Math.pow(p.x-points[m].x,2) +
                    Math.pow(p.y-points[m].y,2)
                );
                if (d < minD) minD = d;
            });
            cost += minD;
        });
        return cost;
    }

    for (var iter = 0; iter < maxIter; iter++) {
        var improved = false;
        var bestCost = totalCost(medoids);

        // Try swapping each medoid with each non-medoid
        for (var mi = 0; mi < k; mi++) {
            for (var pi = 0; pi < n; pi++) {
                if (medoids.indexOf(pi) >= 0) continue;
                var newMedoids = medoids.slice();
                newMedoids[mi] = pi;
                var newCost = totalCost(newMedoids);
                if (newCost < bestCost) {
                    bestCost = newCost;
                    medoids = newMedoids;
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
    return medoids;
}
