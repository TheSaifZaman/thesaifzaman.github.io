Kernel PCA

Hard Time: O(n³) Space: O(n²)

How It Works

Kernel PCA extends standard PCA to capture nonlinear relationships by implicitly mapping data into a higher-dimensional feature space using a kernel function. Instead of computing PCA directly in the high-dimensional space, it operates on the kernel (Gram) matrix, making it computationally tractable.

The kernel trick: instead of explicitly computing the mapping φ(x), we only compute the dot product k(x_i, x_j) = φ(x_i)·φ(x_j). Common kernels include RBF (Gaussian), polynomial, and sigmoid. Eigendecomposition of the centered kernel matrix gives the principal components in the feature space.

Key Properties

  • Nonlinear: Captures complex nonlinear structure in data.
  • Kernel trick: Never explicitly computes high-dimensional mapping.
  • RBF kernel: Maps to infinite-dimensional space.
  • Cost: O(n^3) — more expensive than linear PCA for large n.

function kernelPCA(data, kernel, nComponents) {
    var n = data.length;

    // 1. Compute kernel (Gram) matrix
    var K = [];
    for (var i = 0; i < n; i++) {
        K[i] = [];
        for (var j = 0; j < n; j++) {
            K[i][j] = kernel(data[i], data[j]);
        }
    }

    // 2. Center the kernel matrix
    // K_centered = K - 1_n*K - K*1_n + 1_n*K*1_n
    var rowMeans = K.map(function(row) {
        return row.reduce(function(a,b){return a+b;}) / n;
    });
    var totalMean = rowMeans.reduce(function(a,b){
        return a+b;
    }) / n;
    for (var i2 = 0; i2 < n; i2++) {
        for (var j2 = 0; j2 < n; j2++) {
            K[i2][j2] = K[i2][j2] - rowMeans[i2] -
                rowMeans[j2] + totalMean;
        }
    }

    // 3. Eigendecompose centered kernel matrix
    var eigen = eigenDecompose(K);

    // 4. Project: alphas = eigenvectors / sqrt(eigenvalue)
    var embedding = [];
    for (var i3 = 0; i3 < n; i3++) {
        var coords = [];
        for (var c = 0; c < nComponents; c++) {
            coords.push(
                eigen.vectors[c][i3] *
                Math.sqrt(eigen.values[c])
            );
        }
        embedding.push(coords);
    }
    return embedding;
}

// RBF kernel
function rbfKernel(sigma) {
    return function(a, b) {
        var d2 = 0;
        for (var i = 0; i < a.length; i++) {
            d2 += Math.pow(a[i] - b[i], 2);
        }
        return Math.exp(-d2 / (2 * sigma * sigma));
    };
}
