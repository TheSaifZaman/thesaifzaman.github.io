Latent Dirichlet Allocation

Hard Time: O(n·k·V) Space: O(k·V)

How It Works

LDA is a generative probabilistic model for discovering latent topics in a collection of documents. Each document is modeled as a mixture of topics, and each topic is a distribution over words. The model assumes documents were generated by: (1) choosing a topic mixture, (2) for each word, picking a topic from that mixture, then picking a word from that topic.

Inference is done via Gibbs sampling or variational Bayes. For each word in each document, the algorithm iteratively reassigns the word's topic based on how common the topic is in the document and how common the word is in the topic. After many iterations, the topic-word and document-topic distributions converge.

Key Properties

  • Generative: Models how documents are produced.
  • Soft clustering: Documents belong to multiple topics.
  • Dirichlet prior: Controls topic sparsity (alpha) and word sparsity (beta).
  • Bag-of-words: Ignores word order within documents.

Implementation

function ldaGibbsSampling(docs, K, alpha, beta, nIter) {
    var V = vocabularySize;
    // Initialize random topic assignments
    var z = docs.map(function(doc) {
        return doc.map(function() {
            return Math.floor(Math.random() * K);
        });
    });

    // Count matrices
    var ndk = []; // doc-topic counts
    var nkw = []; // topic-word counts
    var nk = new Array(K).fill(0);

    for (var iter = 0; iter < nIter; iter++) {
        docs.forEach(function(doc, d) {
            doc.forEach(function(w, i) {
                var oldTopic = z[d][i];
                // Decrement counts
                ndk[d][oldTopic]--;
                nkw[oldTopic][w]--;
                nk[oldTopic]--;

                // Sample new topic
                var probs = [];
                for (var k = 0; k < K; k++) {
                    probs[k] = (ndk[d][k] + alpha) *
                        (nkw[k][w] + beta) /
                        (nk[k] + V * beta);
                }
                var newTopic = sampleDiscrete(probs);

                // Increment counts
                z[d][i] = newTopic;
                ndk[d][newTopic]++;
                nkw[newTopic][w]++;
                nk[newTopic]++;
            });
        });
    }
    return { z: z, ndk: ndk, nkw: nkw };
}
