A* Search Algorithm

Hard Time: O(E) Space: O(V)

How It Works

A* is an informed search algorithm that finds the shortest path from a start node to a goal node by combining the actual cost from the start (g-score) with a heuristic estimate to the goal (h-score). It evaluates nodes using f(n) = g(n) + h(n), always expanding the node with the lowest f-score.

The algorithm uses an open set (priority queue) of discovered nodes to be evaluated and a closed set of already-evaluated nodes. At each step, it selects the node with the lowest f-score from the open set, explores its neighbors, and updates their g-scores if a shorter path is found.

A* is guaranteed to find the optimal path if the heuristic is admissible (never overestimates) and consistent (satisfies the triangle inequality). Common heuristics for grids include Manhattan distance and Euclidean distance.

f = g + h
f-score combines actual cost g(n) and heuristic h(n). Expand lowest f-score first.
Admissible Heuristic
Heuristic must never overestimate the true distance to guarantee optimality.
Manhattan Distance
|x1-x2| + |y1-y2| — perfect heuristic for 4-directional grid movement.
Efficiency
Explores far fewer nodes than Dijkstra by using heuristic to guide search toward goal.

Implementation

function aStar(grid, start, end) {
    var rows = grid.length, cols = grid[0].length;
    var dirs = [[0,1],[0,-1],[1,0],[-1,0]];

    function heuristic(r, c) {
        return Math.abs(r - end[0]) + Math.abs(c - end[1]);
    }

    var gScore = [];
    for (var i = 0; i < rows; i++) {
        gScore.push(new Array(cols).fill(Infinity));
    }
    gScore[start[0]][start[1]] = 0;

    var openSet = [[heuristic(start[0], start[1]), start[0], start[1]]];
    var cameFrom = {};
    var closed = {};

    while (openSet.length > 0) {
        openSet.sort(function(a, b) { return a[0] - b[0]; });
        var curr = openSet.shift();
        var cr = curr[1], cc = curr[2];
        var key = cr + ',' + cc;

        if (cr === end[0] && cc === end[1]) {
            return reconstructPath(cameFrom, end);
        }

        if (closed[key]) continue;
        closed[key] = true;

        for (var d = 0; d < dirs.length; d++) {
            var nr = cr + dirs[d][0], nc = cc + dirs[d][1];
            if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
            if (grid[nr][nc] === 1) continue; // wall
            var nKey = nr + ',' + nc;
            if (closed[nKey]) continue;

            var tentG = gScore[cr][cc] + 1;
            if (tentG < gScore[nr][nc]) {
                gScore[nr][nc] = tentG;
                cameFrom[nKey] = key;
                var f = tentG + heuristic(nr, nc);
                openSet.push([f, nr, nc]);
            }
        }
    }
    return null; // no path
}

function reconstructPath(cameFrom, end) {
    var path = [end[0] + ',' + end[1]];
    var curr = end[0] + ',' + end[1];
    while (cameFrom[curr]) {
        curr = cameFrom[curr];
        path.unshift(curr);
    }
    return path;
}
