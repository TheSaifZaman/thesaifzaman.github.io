Travelling Salesman Problem

Hard Time: O(n²·2ⁿ) Space: O(n·2ⁿ)

How It Works

The Travelling Salesman Problem (TSP) asks: given a list of cities and distances between them, find the shortest route that visits every city exactly once and returns to the starting city. TSP is NP-hard — no polynomial-time exact algorithm is known.

The DP + bitmask approach solves TSP in O(n² · 2^n). State dp[mask][i] = minimum cost to visit all cities in the bitmask, ending at city i. Transition: try extending from each subset by adding one more city. This is exponential but tractable for n ≤ 20.

Brute force tries all n! permutations. Approximation algorithms (nearest neighbor, 2-opt) give reasonable solutions in polynomial time for larger instances. The nearest-neighbor heuristic always picks the closest unvisited city.

NP-Hard
No known polynomial exact solution. Exponential algorithms required.
DP + Bitmask
dp[mask][i] = min cost ending at i with visited set = mask. O(n² · 2^n).
Nearest Neighbor
Greedy heuristic: always go to closest unvisited city. Fast but not optimal.
Applications
Route optimization, circuit board drilling, DNA sequencing, logistics.

Implementation

// TSP with DP + Bitmask
function tsp(dist, n) {
    var ALL = (1 << n) - 1;
    var dp = [];
    for (var m = 0; m <= ALL; m++) {
        dp.push(new Array(n).fill(Infinity));
    }
    dp[1][0] = 0; // start at city 0

    for (var mask = 1; mask <= ALL; mask++) {
        for (var u = 0; u < n; u++) {
            if (dp[mask][u] === Infinity) continue;
            if (!(mask & (1 << u))) continue;

            for (var v = 0; v < n; v++) {
                if (mask & (1 << v)) continue; // already visited
                var newMask = mask | (1 << v);
                var cost = dp[mask][u] + dist[u][v];
                if (cost < dp[newMask][v]) {
                    dp[newMask][v] = cost;
                }
            }
        }
    }

    // Find minimum cost to return to start
    var minCost = Infinity;
    for (var i = 1; i < n; i++) {
        var total = dp[ALL][i] + dist[i][0];
        minCost = Math.min(minCost, total);
    }
    return minCost;
}

// Nearest Neighbor Heuristic
function nearestNeighbor(dist, n) {
    var visited = new Array(n).fill(false);
    var path = [0];
    visited[0] = true;
    var total = 0;

    for (var step = 1; step < n; step++) {
        var last = path[path.length - 1];
        var best = -1, bestDist = Infinity;
        for (var j = 0; j < n; j++) {
            if (!visited[j] && dist[last][j] < bestDist) {
                bestDist = dist[last][j];
                best = j;
            }
        }
        visited[best] = true;
        path.push(best);
        total += bestDist;
    }
    total += dist[path[path.length - 1]][0];
    return { path: path, cost: total };
}
