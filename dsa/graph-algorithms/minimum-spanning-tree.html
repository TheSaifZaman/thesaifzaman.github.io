Minimum Spanning Tree (Kruskal/Prim)

Medium Time: O(E log V) Space: O(V+E)

How It Works

A Minimum Spanning Tree (MST) connects all vertices in an undirected, weighted graph with the minimum total edge weight, using exactly V-1 edges and forming no cycles.

Kruskal's Algorithm: Sort all edges by weight. Iterate through edges in order; add an edge to the MST if it doesn't create a cycle (checked using Union-Find). Stop when V-1 edges are added.

Prim's Algorithm: Start from any vertex. Maintain a priority queue of edges connecting the MST to non-MST vertices. Repeatedly extract the minimum-weight edge and add the new vertex to the MST, pushing its edges into the queue.

Kruskal's
Sort edges, use Union-Find. Best for sparse graphs. O(E log E) time.
Prim's
Grow MST from a vertex using min-heap. Best for dense graphs. O(E log V) time.
Cut Property
The lightest edge crossing any cut must be in the MST.
Applications
Network design, clustering, approximation algorithms for NP-hard problems.

// Kruskal's Algorithm with Union-Find
function kruskal(n, edges) {
    edges.sort(function(a, b) { return a[2] - b[2]; });
    var parent = [], rank = [];
    for (var i = 0; i < n; i++) { parent.push(i); rank.push(0); }

    function find(x) {
        if (parent[x] !== x) parent[x] = find(parent[x]);
        return parent[x];
    }
    function union(a, b) {
        var ra = find(a), rb = find(b);
        if (ra === rb) return false;
        if (rank[ra] < rank[rb]) { var t = ra; ra = rb; rb = t; }
        parent[rb] = ra;
        if (rank[ra] === rank[rb]) rank[ra]++;
        return true;
    }

    var mst = [], totalWeight = 0;
    for (var e = 0; e < edges.length; e++) {
        if (union(edges[e][0], edges[e][1])) {
            mst.push(edges[e]);
            totalWeight += edges[e][2];
            if (mst.length === n - 1) break;
        }
    }
    return { mst: mst, totalWeight: totalWeight };
}

// Prim's Algorithm
function prim(n, adj) {
    var inMST = new Array(n).fill(false);
    var key = new Array(n).fill(Infinity);
    key[0] = 0;
    var pq = [[0, 0]]; // [weight, vertex]
    var totalWeight = 0;

    while (pq.length > 0) {
        pq.sort(function(a, b) { return a[0] - b[0]; });
        var curr = pq.shift();
        var u = curr[1];
        if (inMST[u]) continue;
        inMST[u] = true;
        totalWeight += curr[0];
        for (var i = 0; i < adj[u].length; i++) {
            var v = adj[u][i][0], w = adj[u][i][1];
            if (!inMST[v] && w < key[v]) {
                key[v] = w;
                pq.push([w, v]);
            }
        }
    }
    return totalWeight;
}
