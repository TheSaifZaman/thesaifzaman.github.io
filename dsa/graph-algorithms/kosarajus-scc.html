Kosaraju's SCC

Hard Time: O(V+E) Space: O(V)

How It Works

Kosaraju's algorithm finds all strongly connected components using two DFS passes. It leverages the fact that a graph and its transpose have the same SCCs.

Pass 1: Perform DFS on the original graph, pushing vertices to a stack in order of finish time (post-order). Pass 2: Build the transpose graph (reverse all edges). Pop vertices from the stack and perform DFS on the transpose — each DFS tree forms one SCC.

The intuition is that the finish-time ordering ensures that in Pass 2, we process SCC roots before their descendants in the condensation DAG. In the transposed graph, we can only reach vertices in the same SCC from an SCC root, so each DFS in Pass 2 exactly captures one SCC.

Two Passes
First DFS orders by finish time; second DFS on transpose finds SCCs.
Transpose Graph
Reverse all edge directions. Same SCCs, but cross-SCC edges are reversed.
Finish Time Order
Processing in reverse finish order ensures SCC roots are visited first.
Conceptually Simpler
Easier to understand than Tarjan's, but requires two full passes.

Implementation

function kosarajuSCC(n, adj) {
    var visited = new Array(n).fill(false);
    var order = []; // finish-time order

    // Pass 1: DFS on original graph
    function dfs1(u) {
        visited[u] = true;
        for (var i = 0; i < adj[u].length; i++) {
            if (!visited[adj[u][i]]) dfs1(adj[u][i]);
        }
        order.push(u);
    }

    for (var i = 0; i < n; i++) {
        if (!visited[i]) dfs1(i);
    }

    // Build transpose
    var radj = [];
    for (var i = 0; i < n; i++) radj.push([]);
    for (var u = 0; u < n; u++) {
        for (var j = 0; j < adj[u].length; j++) {
            radj[adj[u][j]].push(u);
        }
    }

    // Pass 2: DFS on transpose in reverse finish order
    visited = new Array(n).fill(false);
    var sccs = [];

    function dfs2(u, scc) {
        visited[u] = true;
        scc.push(u);
        for (var i = 0; i < radj[u].length; i++) {
            if (!visited[radj[u][i]]) dfs2(radj[u][i], scc);
        }
    }

    for (var k = n - 1; k >= 0; k--) {
        var v = order[k];
        if (!visited[v]) {
            var scc = [];
            dfs2(v, scc);
            sccs.push(scc);
        }
    }
    return sccs;
}
